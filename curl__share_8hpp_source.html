137 /**
138  * @tparam Shared_mutex_t For shared or single lock, their unlock function must be
139  * the same function -- Shared_mutex_t::unlock().
140  *
141  * If Shared_mutex_t lock, lock_shared or unlock
142  * throw an exception, it would terminte the program.
143  *
144  * If Shared_mutex_t has type Ret_except_t (Ret == void), then
145  * Shared_mutex_t(Ret_except_t&) would be called.
146  *
147  * Pass void to disable locking, which make
148  * multithreaded use unsafe.
149  */
150 template <class Shared_mutex_t = utils::shared_mutex>
151 class Share: public Share_base {
152  static constexpr const std::size_t mutex_num = 5;
153 
154  Shared_mutex_t mutexes[mutex_num];
155 
156  auto get_mutex(Options option) noexcept -> Shared_mutex_t*
157  {
158  switch (option) {
159  case Options::none:
160  return nullptr;
161  case Options::cookie:
162  return &mutexes[0];
163  case Options::dns:
164  return &mutexes[1];
165  case Options::ssl_session:
166  return &mutexes[2];
167  case Options::connection_cache:
168  return &mutexes[3];
169  case Options::psl:
170  return &mutexes[4];
171  default:
172  return nullptr;
173  }
174  }
175 
176 public:
177  using Share_base::Share_base;
178 
179  void enable_multithreaded_share() noexcept
180  {
181  add_lock([](CURL *handle, curl_lock_data data, curl_lock_access access, void *userptr) noexcept {
182  auto &share = *static_cast<Share*>(userptr);
183  auto *mutex = share.get_mutex(static_cast<Share_base::Options>(data));
184 
185  if (!mutex)
186  return;
187 
188  if (access == CURL_LOCK_ACCESS_SHARED)
189  mutex->lock_shared();
190  else
191  mutex->lock();
192  }, [](CURL *handle, curl_lock_data data, void *userptr) noexcept {
193  auto &share = *static_cast<Share*>(userptr);
194  auto *mutex = share.get_mutex(static_cast<Share_base::Options>(data));
195 
196  if (mutex)
197  mutex->unlock();
198  }, this);
199  }
200  void disable_multithreaded_share() noexcept
201  {
202  add_lock(nullptr, nullptr, nullptr);
203  }
204 };
