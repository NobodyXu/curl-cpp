163 /**
164  * @tparam Shared_mutex_t For shared or single lock, their unlock function must be
165  * the same function -- Shared_mutex_t::unlock().
166  *
167  * If Shared_mutex_t lock, lock_shared or unlock
168  * throw an exception, it would terminte the program.
169  *
170  * If Shared_mutex_t has type Ret_except_t (Ret == void), then
171  * Shared_mutex_t(Ret_except_t&) would be called.
172  *
173  * Pass void to disable locking, which make
174  * multithreaded use unsafe.
175  */
176 template <class Shared_mutex_t = utils::shared_mutex>
177 class Share: public Share_base {
178  static constexpr const std::size_t mutex_num = 5;
179 
180  Shared_mutex_t mutexes[mutex_num];
181 
182  auto get_mutex(Options option) noexcept -> Shared_mutex_t*
183  {
184  switch (option) {
185  case Options::none:
186  return nullptr;
187  case Options::cookie:
188  return &mutexes[0];
189  case Options::dns:
190  return &mutexes[1];
191  case Options::ssl_session:
192  return &mutexes[2];
193  case Options::connection_cache:
194  return &mutexes[3];
195  case Options::psl:
196  return &mutexes[4];
197  default:
198  return nullptr;
199  }
200  }

212  void enable_multithreaded_share() noexcept
213  {
214  add_lock([](CURL *handle, curl_lock_data data, curl_lock_access access, void *userptr) noexcept {
215  auto &share = *static_cast<Share*>(userptr);
216  auto *mutex = share.get_mutex(static_cast<Share_base::Options>(data));
217 
218  if (!mutex)
219  return;
220 
221  if (access == CURL_LOCK_ACCESS_SHARED)
222  mutex->lock_shared();
223  else
224  mutex->lock();
225  }, [](CURL *handle, curl_lock_data data, void *userptr) noexcept {
226  auto &share = *static_cast<Share*>(userptr);
227  auto *mutex = share.get_mutex(static_cast<Share_base::Options>(data));
228 
229  if (mutex)
230  mutex->unlock();
231  }, this);
232  }
233  void disable_multithreaded_share() noexcept
234  {
235  add_lock(nullptr, nullptr, nullptr);
236  }
