259 auto Easy_ref_t::set_pause(PauseOptions option) noexcept -> Ret_except<code, std::bad_alloc, Exception>
260 {
261  int bitmask = 0;
262  if (option & PauseOptions::recv)
263  bitmask |= CURLPAUSE_RECV;
264  if (option & PauseOptions::send)
265  bitmask |= CURLPAUSE_SEND;
266 
267  auto result = curl_easy_pause(curl_easy, bitmask);
268  if (result == CURLE_WRITE_ERROR)
269  return {code::writeback_error};
270  else if (result == CURLE_OUT_OF_MEMORY)
271  return {std::bad_alloc{}};
272  else if (result != CURLE_OK)
273  return {Exception{result}};
274  else
275  return {code::ok};
